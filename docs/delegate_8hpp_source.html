501 template <typename Ret, typename... Args>
502 template <typename F, std::enable_if_t<!std::is_same_v<
503 std::decay_t<F>, delegate<Ret(Args...)>>> *>
504 void delegate<Ret(Args...)>::bind(F &&f) {
505 static_assert(
506 std::is_invocable_r_v<Ret, decltype(f), Args...>,
507 "The function object must have a call signature of Ret(Args...)");
508 reset();
509 if constexpr (sizeof(std::decay_t<F>) <= 16) {
510 // store the f inline with placement new into storage.
511 new (&storage) std::decay_t<F>(std::forward<F>(f));
512 table = impl::vtable::template make_inline<std::decay_t<F>>();
513 invoke = &inline_invoke<std::decay_t<F>>;
514 } else {
515 // cannot store inline -> have to use the heap
516 *reinterpret_cast<std::decay_t<F> **>(&storage) =
517 new std::decay_t<F>(std::forward<F>(f));
518 table = impl::vtable::template make_heap<std::decay_t<F>>();
519 invoke = &heap_invoke<std::decay_t<F>>;
520 }
521 }

536 template <typename Ret, typename... Args>
537 Ret delegate<Ret(Args...)>::free_func_invoke(void *object, Args... args) {
538 // storage will contain a function pointer. The void* param will be the
539 // address of storage. This means the real type of the param is 'pointer to
540 // function pointer'.
541 using type = Ret (*)(Args...); // type alias to easily cast
542 // static_cast to Ret to account for the case where Ret = void
543 return static_cast<Ret>((*static_cast<type *>(object))(args...));
544 }

546 template <typename Ret, typename... Args>
547 template <typename T>
548 Ret delegate<Ret(Args...)>::mfn_invoke(void *object, Args... args) {
549 // storage will contain a mfn_holder_t<T, Ret, Args...> object. As such,
550 // object's real type is 'pointer to mfn_holder_t<T, Ret, Args...>'.
551 using type = impl::mfn_holder_t<T, Ret, Args...>;
552 // funky member function invocation through member function pointer.
553 // without the casts, it would look like this: (t->*func)(args), where t is
554 // a pointer to the object and func is the member function pointer.
555 return static_cast<Ret>(
556 (static_cast<type *>(object)->t->*(static_cast<type *>(object)->func))(
557 args...));
558 }

571 template <typename Ret, typename... Args>
572 Ret delegate<Ret(Args...)>::null_invoke(void *, Args...) {
573 // this function does a null invoke, i.e. does nothing. In case Ret != void
574 // and Ret is constructible with no arguments, a statically allocated value
575 // of Ret is returned.
576 if constexpr ((!std::is_same_v<Ret, void>)&&std::is_constructible_v<
577 std::decay_t<Ret>>) {
578 static std::decay_t<Ret> r{};
579 if constexpr (std::is_rvalue_reference_v<Ret>) {
580 return std::move(r);
581 } else
582 return r;
583 }
