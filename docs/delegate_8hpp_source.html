520 template <typename Ret, typename... Args>
521 template <typename F, std::enable_if_t<!std::is_same_v<
522 std::decay_t<F>, delegate<Ret(Args...)>>>*>
523 void delegate<Ret(Args...)>::bind(F&& f) {
524 static_assert(
525 std::is_invocable_r_v<Ret, decltype(f), Args...>,
526 "The function object must have a call signature of Ret(Args...)");
527 reset();
528 if constexpr (sizeof(std::decay_t<F>) <= 16) {
529 // store the f inline with placement new into storage.
530 new (&storage) std::decay_t<F>(std::forward<F>(f));
531 table = impl::vtable::template make_inline<std::decay_t<F>>();
532 invoke = &inline_invoke<std::decay_t<F>>;
533 } else {
534 // cannot store inline -> have to use the heap
535 *reinterpret_cast<std::decay_t<F>**>(&storage) =
536 new std::decay_t<F>(std::forward<F>(f));
537 table = impl::vtable::template make_heap<std::decay_t<F>>();
538 invoke = &heap_invoke<std::decay_t<F>>;
539 }

555 template <typename Ret, typename... Args>
556 Ret delegate<Ret(Args...)>::free_func_invoke(void* object, Args... args) {
557 // storage will contain a function pointer. The void* param will be the
558 // address of storage. This means the real type of the param is 'pointer to
559 // function pointer'.
560 using type = Ret (*)(Args...); // type alias to easily cast
561 // static_cast to Ret to account for the case where Ret = void. Else there
562 // would be a compile error.
563 return static_cast<Ret>((*static_cast<type*>(object))(args...));
564 }

566 template <typename Ret, typename... Args>
567 template <typename T>
568 Ret delegate<Ret(Args...)>::mfn_invoke(void* object, Args... args) {
569 // storage will contain a mfn_holder_t<T, Ret, Args...> instance inline. As
570 // such, object's real type is 'pointer to mfn_holder_t<T, Ret, Args...>'.
571 using type = impl::mfn_holder_t<T, Ret, Args...>;
572 // funky member function invocation through member function pointer.
573 // without the casts, it would look like this: (t->*func)(args...), where t
574 // is a pointer to the instance and func is the member function pointer of
575 // t.
576 return static_cast<Ret>(
577 (static_cast<type*>(object)->t->*(static_cast<type*>(object)->func))(
578 args...));
579 }

649 template <typename T>
650 void impl::vtable::heap_copy(void* dest, const void* source) {
651 // when copying, we have to allocate a new function object of type T.
652 // source is the this delegates's storage's address. The storage contains a
653 // pointer to T. As such, the correct type for source is 'pointer to pointer
654 // to T'.
655 T* t = new T(*(*static_cast<const T* const *>(source)));
656 std::memcpy(dest, &t, sizeof(T*));
657 }
